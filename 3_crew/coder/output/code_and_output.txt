def calculate_series(n):
    total = 0.0
    for i in range(n):
        term = 1 / (2 * i + 1)  # Calculate the term (1/1, 1/3, 1/5, ...)
        if i % 2 == 0:
            total += term  # Add the term if i is even
        else:
            total -= term  # Subtract the term if i is odd
    return total * 4

# Calculate the sum of the first 10000 terms
result = calculate_series(10000)
print(result)
